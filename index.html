(function () {

  var JIRA_BASE = 'http://localhost:3000/jira/mcp';

  var activeTeam = 'all';

  var selCard = null;

  var profId = null;

  var profName = null;

  var profApps = [];

  var ticketKey = null;

  var hireName = null;


  function el(id) { return document.getElementById(id); }

  function lock(id) { el(id).classList.add('locked'); }

  function unlock(id) { el(id).classList.remove('locked'); }


  // ── Jira via MCP fetch ──

  function jiraSearch(name, onSuccess, onError) {

    var jql = 'project = GSSD AND summary ~ "New Prep Request" AND summary ~ "' + name + '" ORDER BY created DESC';

    fetch(JIRA_BASE + '/rest/api/2/search?jql=' + encodeURIComponent(jql) + '&fields=summary,created&maxResults=10', {

      headers: { 'Accept': 'application/json' }

    })

    .then(function(r) { return r.json(); })

    .then(function(data) {

      var results = (data.issues || []).map(function(i) {

        return { key: i.key, summary: i.fields.summary, created: (i.fields.created || '').slice(0,10) };

      });

      onSuccess(results);

    })

    .catch(onError);

  }


  function jiraComment(key, body, onSuccess, onError) {

    fetch(JIRA_BASE + '/rest/api/2/issue/' + key + '/comment', {

      method: 'POST',

      headers: { 'Content-Type': 'application/json', 'Accept': 'application/json' },

      body: JSON.stringify({ body: body })

    })

    .then(function(r) {

      if (!r.ok) return r.json().then(function(e) { throw new Error(e.errorMessages ? e.errorMessages.join(', ') : 'Status ' + r.status); });

      return r.json();

    })

    .then(onSuccess)

    .catch(onError);

  }


  // ── Filters ──

  el('searchInput').addEventListener('input', applyFilters);

  document.querySelectorAll('.filter-btn').forEach(function(btn) {

    btn.addEventListener('click', function() {

      activeTeam = this.dataset.team;

      document.querySelectorAll('.filter-btn').forEach(function(b) { b.classList.remove('active'); });

      this.classList.add('active');

      applyFilters();

    });

  });


  function applyFilters() {

    var q = el('searchInput').value.toLowerCase().trim();

    var cards = document.querySelectorAll('.role-card[data-team]');

    var visible = 0;

    cards.forEach(function(c) {

      var teamOk = activeTeam === 'all' || c.dataset.team === activeTeam;

      var searchOk = !q || c.dataset.text.includes(q) || c.dataset.apps.includes(q) ||

        c.dataset.team.toLowerCase().includes(q) ||

        c.querySelector('.role-name').textContent.toLowerCase().includes(q) ||

        c.querySelector('.profile-id').textContent.toLowerCase().includes(q);

      if (teamOk && searchOk) { c.classList.remove('hidden'); visible++; }

      else c.classList.add('hidden');

    });

    el('resultsCount').textContent = visible + ' profile' + (visible !== 1 ? 's' : '');

    el('heroCount').textContent = visible;

    el('emptyState').style.display = visible === 0 ? 'block' : 'none';

  }


  // ── Step 1 ──

  document.querySelectorAll('.select-btn').forEach(function(btn) {

    btn.addEventListener('click', function() {

      var card = this.closest('.role-card');

      if (selCard) { selCard.classList.remove('selected'); selCard.querySelector('.select-btn').textContent = 'Select Profile'; }

      selCard = card; card.classList.add('selected'); this.textContent = '✓ Selected';

      profId = card.dataset.id; profName = card.dataset.name; profApps = card.dataset.apps.split(' ');

      el('pill2id').textContent = profId; el('pill2name').textContent = profName;

      el('pill2').classList.add('show'); el('hint2').textContent = profId + ' · ' + profName;

      unlock('panel2'); refreshToggles();

      lock('panel3'); el('hint3').textContent = 'Complete Steps 1 & 2 to unlock';

      el('pill3').classList.remove('show'); resetHire();

      setTimeout(function() { el('panel2').scrollIntoView({ behavior: 'smooth', block: 'start' }); }, 100);

    });

  });


  el('pill2change').addEventListener('click', function() {

    if (selCard) { selCard.classList.remove('selected'); selCard.querySelector('.select-btn').textContent = 'Select Profile'; }

    selCard = null; profId = null; profName = null; profApps = [];

    lock('panel2'); el('pill2').classList.remove('show');

    lock('panel3'); el('pill3').classList.remove('show');

    resetHire(); window.scrollTo({ top: 0, behavior: 'smooth' });

  });


  function refreshToggles() {

    document.querySelectorAll('#appToggles .app-toggle').forEach(function(t) {

      var key = t.dataset.app.toLowerCase().replace(/ /g, '-');

      t.classList.remove('on', 'included'); t.querySelector('.chk').textContent = '';

      if (profApps.includes(key)) { t.classList.add('included'); t.querySelector('.chk').textContent = ' ✓'; }

    });

  }


  document.querySelectorAll('#appToggles .app-toggle').forEach(function(t) {

    t.addEventListener('click', function() {

      if (this.classList.contains('included')) return;

      this.classList.toggle('on');

      this.querySelector('.chk').textContent = this.classList.contains('on') ? ' ✓' : '';

    });

  });


  // ── Step 2 ──

  el('continueBtn').addEventListener('click', function() {

    el('pill3id').textContent = profId; el('pill3name').textContent = profName;

    el('pill3').classList.add('show'); el('hint3').textContent = profId + ' · ' + profName;

    unlock('panel3');

    setTimeout(function() { el('panel3').scrollIntoView({ behavior: 'smooth', block: 'start' }); }, 100);

  });


  el('pill3change').addEventListener('click', function() {

    lock('panel3'); el('pill3').classList.remove('show'); resetHire();

  });


  // ── Step 3 ──

  el('searchBtn').addEventListener('click', doSearch);

  el('hireInput').addEventListener('keydown', function(e) { if (e.key === 'Enter') doSearch(); });


  function doSearch() {

    var q = el('hireInput').value.trim();

    if (!q) return;

    var btn = el('searchBtn');

    btn.querySelector('.mini-spin').style.display = 'block';

    btn.classList.add('loading'); btn.disabled = true;

    var res = el('hireResults');

    res.innerHTML = '<div class="hire-results-msg">Searching Jira…</div>';

    res.classList.add('show');


    jiraSearch(q, function(results) {

      btn.querySelector('.mini-spin').style.display = 'none';

      btn.classList.remove('loading'); btn.disabled = false;

      if (!results.length) {

        res.innerHTML = '<div class="hire-results-msg">No "New Prep Request" tickets found for "' + q + '". The Workday submission may still be processing (~1 hour).</div>';

      } else {

        res.innerHTML = results.map(function(h) {

          return '<div class="hire-result-row" data-key="' + h.key + '" data-summary="' + h.summary.replace(/"/g, '&quot;') + '">' +

            '<div><div class="hire-result-name">' + h.summary + '</div><div class="hire-result-date">Created ' + h.created + '</div></div>' +

            '<span class="hire-result-key">' + h.key + '</span></div>';

        }).join('');

        res.querySelectorAll('.hire-result-row').forEach(function(row) {

          row.addEventListener('click', function() { confirmHire(this.dataset.key, this.dataset.summary); });

        });

      }

    }, function(err) {

      btn.querySelector('.mini-spin').style.display = 'none';

      btn.classList.remove('loading'); btn.disabled = false;

      res.innerHTML = '<div class="hire-results-msg">Could not reach Jira. (' + err.message + ')</div>';

    });

  }


  function confirmHire(key, name) {

    ticketKey = key; hireName = name;

    el('lookupSection').style.display = 'none';

    el('hireResults').classList.remove('show');

    el('confirmedName').textContent = name;

    el('confirmedKey').textContent = '🎫 ' + key + ' · GSSD ticket found';

    el('confirmedHire').classList.add('show');

    el('submitSection').style.display = 'block';

  }


  el('changeHireBtn').addEventListener('click', resetHire);


  function resetHire() {

    ticketKey = null; hireName = null;

    el('lookupSection').style.display = 'block';

    el('hireInput').value = '';

    el('hireResults').classList.remove('show'); el('hireResults').innerHTML = '';

    el('confirmedHire').classList.remove('show');

    el('submitSection').style.display = 'none';

    el('toast').classList.remove('show'); el('toastErr').classList.remove('show');

  }


  // ── Step 4 ──

  el('submitBtn').addEventListener('click', function() {

    if (!ticketKey) return;

    var notes = el('notesInput').value.trim();

    var extra = [];

    document.querySelectorAll('#appToggles .app-toggle.on').forEach(function(t) { extra.push(t.dataset.app); });

    var stdApps = profApps.map(function(a) { return a.replace(/-/g,' ').replace(/\b\w/g, function(l) { return l.toUpperCase(); }); }).join(', ');

    var btn = el('submitBtn');

    btn.classList.add('loading'); btn.disabled = true;

    el('toast').classList.remove('show'); el('toastErr').classList.remove('show');


    var body =

      '[External Onboarding Access Update]\n\n' +

      'New Hire: ' + hireName + '\n' +

      'Role Profile: ' + profId + ' - ' + profName + '\n' +

      'Standard Apps: ' + stdApps + '\n' +

      'Additional Apps Requested: ' + (extra.length ? extra.join(', ') : 'None') + '\n' +

      'Notes: ' + (notes || 'None') + '\n\n' +

      'Submitted via Sonos External Onboarding Portal';


    jiraComment(ticketKey, body, function() {

      var url = 'https://jira.sonos.com/browse/' + ticketKey;

      el('toastMsg').innerHTML =

        'Comment posted to <a href="' + url + '" target="_blank"><strong>' + ticketKey + '</strong></a>. IT Ops will provision access shortly.<br><br>' +

        '<strong>Hire:</strong> ' + hireName + '<br>' +

        '<strong>Profile:</strong> ' + profId + ' \u2013 ' + profName + '<br>' +

        '<strong>Standard Apps:</strong> ' + stdApps + '<br>' +

        '<strong>Extra Apps:</strong> ' + (extra.length ? extra.join(', ') : 'None') + '<br>' +

        '<strong>Notes:</strong> ' + (notes || 'None');

      el('toast').classList.add('show');

      el('notesInput').value = '';

      document.querySelectorAll('#appToggles .app-toggle.on').forEach(function(t) {

        t.classList.remove('on'); t.querySelector('.chk').textContent = '';

      });

      btn.classList.remove('loading'); btn.disabled = false;

    }, function(err) {

      el('toastErr').classList.add('show');

      el('toastErrMsg').textContent = 'Could not post to Jira: ' + err.message + '. Contact IT Ops on #it-ops.';

      btn.classList.remove('loading'); btn.disabled = false;

    });

  });


})();
